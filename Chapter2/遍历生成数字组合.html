第五节 数字递增形式的组合 

n,k = 2,4

arr = [0]*k

while True:
    # print 
    print(*arr,sep="")
    index = k-1
    # find the index that will be dealed
    while index>=0 and arr[index] == n-1:
        index -= 1
    # find all result
    if index < 0:
        break;
    arr[index] += 1

    # 处理进制
    for i in range(index+1,len(arr)):
        arr[i] = 0
